 54    def burst_dict(self, burst: str):
 55        """Builds a dictionary as an internal representation of a burst.
 56
 57        Args:
 58            burst (str): The burst string that should be turned into a dictionary.
 59
 60        Returns:
 61            dict: The computed dictionary with soliton number as key and a list containing exterior nodes and entry time as value.
 62        """
 63        burst_copy = copy.copy(burst)
 64        burst_dict = {}
 65        entry = 0 # entry time of the current soliton, is increased every time there is another soliton with a k_i
 66        soliton = 1 # number of the current soliton
 67        first = re.search(r"[(][0-9]+[,][0-9]+[)]", burst_copy) # first soliton in burst
 68        nodes = re.findall(r"[0-9]+", first.group()) # find the two exterior nodes
 69        burst_dict[soliton] = [self.soliton_graph.exterior_nodes_reverse[nodes[0]], self.soliton_graph.exterior_nodes_reverse[nodes[1]], 0] # first soliton entries at time step 0 (take node ids, not node labels)
 70        burst_copy = burst_copy[first.span()[1]:] # remove all the information on the first soliton from the burst
 71        while burst_copy != "":
 72            found = re.search(r"[||][0-9]+[(][0-9]+[,][0-9]+[)]", burst_copy)
 73            all_nums = re.findall(r"[0-9]+", found.group())
 74            entry = entry + int(all_nums[0]) # add the current k_i to the entry time
 75            soliton += 1
 76            burst_dict[soliton] = [self.soliton_graph.exterior_nodes_reverse[all_nums[1]], self.soliton_graph.exterior_nodes_reverse[all_nums[2]], entry] # take node ids, not node labels
 77            burst_copy = burst_copy[found.span()[1]:] # remove from burst
 78            
 79        return burst_dict

122    def find_all_travs_given_burst (self, graph: nx.Graph, burst_dict: dict, t: int, s_pos_all_timesteps: list, bindings_all_timesteps: list, binds_all_timesteps: list, poss_suc_combs_all_timesteps: list, travs: list):
123        """Finds all possible traversals for a given burst by using a recursive backtracking algorithm.
124
125        Args:
126            graph (nx.Graph): Graph the traversals should be found in.
127            burst_dict (dict): The burst that is used. 
128            t (int): Current timestep.
129            s_pos_all_timesteps (list): Soliton positions for all past timesteps.
130            bindings_all_timesteps (list): Binding types of all edges in the graph for all past timesteps.
131            binds_all_timesteps (list): Binding type of the last edge that was traversed by each soliton for all past timesteps.
132            poss_suc_combs_all_timesteps (list): Possible combinations of successor nodes for all past timesteps.
133            travs (list): All currently found traversals.
134
135        Returns:
136            list: All found traversals (is empty if no traversal exists).
137        """
138        t += 1
139        akt_bindings = bindings_all_timesteps[t-1]
140        akt_binds = binds_all_timesteps[t-1]
141        akt_positions = s_pos_all_timesteps[t-1]
142
143        # in this timestep find possible edges for each soliton individually
144        possible_edges = {} # possible edges for each soliton in this timestep
145        possible_nodes = {} # possible nodes for each soliton in this timestep
146        for soliton in burst_dict:
147            end = burst_dict[soliton][1] # end node for this soliton
148            akt_pos = akt_positions[soliton] # current position of this soliton
149            possible_edges[soliton] = []
150            possible_nodes[soliton] = []
151            if akt_pos == -2 or (akt_pos == end and s_pos_all_timesteps[t-2][soliton] != -1 and t > 1): # if soliton already traversed the graph successfully
152                possible_edges[soliton].append(-2)
153                possible_nodes[soliton].append(-2)
154            elif burst_dict[soliton][2] > t: # if soliton's entry time has not come yet
155                possible_edges[soliton].append(-1)
156                possible_nodes[soliton].append(-1)
157            elif burst_dict[soliton][2] == t: # if soliton enters the graph now (place it at its starting node)
158                possible_edges[soliton].append(burst_dict[soliton][0])
159                possible_nodes[soliton].append(burst_dict[soliton][0])
160            else:
161                for node in list(nx.neighbors(graph, akt_pos)): # iterate over all nodes that are adjacent to current position
162                    # soliton can not go to an exterior node that is not the end node, soliton is not allowed to make a direct turnaround and edge to next node has to have the right binding type
163                    if ((node not in self.soliton_graph.exterior_nodes) or (node == end)) and node != s_pos_all_timesteps[t-2][soliton] and akt_bindings[tuple(sorted((akt_pos, node)))] != akt_binds[soliton]:
164                        possible_edges[soliton].append(tuple(sorted((akt_pos, node))))
165                        possible_nodes[soliton].append(node)
166        # find all possible combinations of the found edges 
167        edges_combs = []
168        nodes_combs = []
169        for soliton in possible_edges:
170            if possible_edges[soliton] == []: # if for one soliton there are no possible next edges, no combinations are possible
171                edges_combs.clear()
172                nodes_combs.clear()
173            if soliton != 1 and not any(edges_combs): # if no possible edge combination exists anymore, terminate the loop (not any(edges_combs) makes sure list is empty or only contains empty lists)
174                break
175            for i, edge in enumerate(possible_edges[soliton]): # for all possible edges for this soliton
176                if soliton == 1:
177                    edges_combs.append([edge]) # simply add the edge if it's the first soliton
178                    nodes_combs.append({soliton: possible_nodes[soliton][i]}) # add dictionary with node for this soliton
179                else:
180                    for j, comb in enumerate(edges_combs): # if it's not the first soliton loop over all existing combinations
181                        if edge == -1 or edge == -2 or edge not in comb: # if edge is not already in combination (or soliton is outside the graph)
182                            if len(comb) < soliton:
183                                edges_combs[j].append(edge) # add the edge if this combination doesn't contain an edge for this soliton already
184                                nodes_combs[j][soliton] = possible_nodes[soliton][i]
185                            else: # else copy the combination, remove the last edge (the edge for this soliton), add the edge and add this new combination
186                                comb_copy = copy.deepcopy(comb)
187                                comb_copy.pop()
188                                comb_copy.append(edge)
189                                edges_combs.append(comb_copy)
190                                node_comb_copy = copy.deepcopy(nodes_combs[j])
191                                node_comb_copy[soliton] = possible_nodes[soliton][i] # overwrite
192                                nodes_combs.append(node_comb_copy)
193                        if i == len(possible_edges[soliton])-1: # if we are looking at the last possible edge of this soliton
194                            if len(comb) < soliton: # and the combination does not contain an edge for each soliton
195                                edges_combs[j] = [] # "delete" the combination
196                                nodes_combs[j] = {}
197        edges_combs = [elem for elem in edges_combs if elem != []]
198        nodes_combs = [elem for elem in nodes_combs if elem]
199
200        finished = True
201        for soliton in akt_positions:
202            if (akt_positions[soliton] != burst_dict[soliton][1] and akt_positions[soliton] != -2) or s_pos_all_timesteps[t-2][soliton] == -1 or t < 2: # if not all solitons reached their end node or already left the graph
203                finished = False
204                break
205        # base case 1: if all solitons traversed the graph successfully
206        if finished == True:
207            trav = []
208            for i, pos in enumerate(s_pos_all_timesteps):
209                this_timestep = (pos, bindings_all_timesteps[i]) # for each timestep, add a tuple of the solitons' positions and the bindings
210                trav.append(this_timestep)
211            travs.append(trav) # trav = traversal of all solitons through the graph
212            return travs
213
214        # base case 2: if we have two successor-equivalent configurations in the configuration trail
215        count = 1
216        for k in range(0, len(bindings_all_timesteps)-1):
217            if akt_bindings == bindings_all_timesteps[k] and akt_positions == s_pos_all_timesteps[k] and nodes_combs == poss_suc_combs_all_timesteps[k] and -1 not in nodes_combs[0].values(): # if we already had that exact graph, position map and successor position combinations (and all solitons already entered the graph) in this configuration trail
218                count += 1
219                if count == 2:
220                    trav = []
221                    for i, pos in enumerate(s_pos_all_timesteps):
222                        this_timestep = (pos, bindings_all_timesteps[i])
223                        trav.append(this_timestep)
224                    travs.append([trav[:k+1], k])
225                    return travs
226
227        # iterate over all possible combinations
228        for i, comb in enumerate(nodes_combs):
229            s_pos_all_timesteps.append(comb)
230            bindings = copy.deepcopy(akt_bindings)
231            binds = copy.deepcopy(akt_binds)
232            for j, edge in enumerate(edges_combs[i]): # for all edges in chosen combination
233                if isinstance(edge, tuple): # binds and bindings only change if "edge" is actually an edge (if edge is -1, -2 or the start node, no edge is traversed)
234                    binds[j+1] = bindings[tuple(sorted(edge))] # change bind to binding type of edge that was just traversed
235                    bindings = self.change_bindings(bindings, tuple(sorted(edge))) # change bindings
236            bindings_all_timesteps.append(bindings)
237            binds_all_timesteps.append(binds)
238            poss_suc_combs_all_timesteps.append(nodes_combs)
239            # call function recursively: if eventually all solitons reach their end node if we go further with the decision we just made then travs is changed (new traversal added)
240            travs = self.find_all_travs_given_burst(graph, burst_dict, t, s_pos_all_timesteps, bindings_all_timesteps, binds_all_timesteps, poss_suc_combs_all_timesteps, travs)
241            # try different decisions next, so we need variables in the state before the last decision
242            s_pos_all_timesteps.pop()
243            bindings_all_timesteps.pop()
244            binds_all_timesteps.pop()
245            poss_suc_combs_all_timesteps.pop()
246
247        return travs # if at some point no other traversal could be added, then all possible traversals are found

250    def call_find_all_travs_given_burst (self, burst_dict: dict, soliton_graph: SolitonGraph):
251        """Initializes some parameters and then calls `find_all_travs_given_burst` with them.
252
253        Args:
254            burst_dict (dict): The burst that is used. 
255            soliton_graph (SolitonGraph): Soliton graph the traversals should be found in.
256
257        Returns:
258            list: All found travs as traversals (returns empty list when no traversal is found).
259        """
260        soliton_graph_copy = copy.deepcopy(soliton_graph) # working on copy of graph so no unwanted changes are made
261        graph = soliton_graph_copy.graph
262        t = 0
263        bindings_all_timesteps = [soliton_graph_copy.bindings] # already add bindings for timestep 0
264        soliton_positions = {}
265        binds = {}
266        for soliton in burst_dict:
267            binds[soliton] = 0
268            if burst_dict[soliton][2] == 0: # if soliton enters the graph at timestep 0
269                soliton_positions[soliton] = burst_dict[soliton][0] # starts at its start node
270            else:
271                soliton_positions[soliton] = -1 # soliton is not in graph at timestep 0
272        # already add positions and binds for timestep 0
273        s_pos_all_timesteps = [soliton_positions]
274        binds_all_timesteps = [binds]
275        poss_suc_combs_all_timesteps = []
276        travs = []
277
278        travs = self.find_all_travs_given_burst(graph, burst_dict, t, s_pos_all_timesteps, bindings_all_timesteps, binds_all_timesteps, poss_suc_combs_all_timesteps, travs)
279        traversals = self.build_traversals(travs, soliton_graph)
280
281        return traversals

284    def all_traversals (self):
285        """Calls `call_find_all_travs_given_burst` for all states of the automaton and all bursts in order to get all possible traversals in all states.
286
287        Returns:
288            bool: Whether the soliton automaton is deterministic or not.
289            bool: Whether the soliton automaton is strongly deterministic or not.
290            bool: Whether the soliton automaton is reachability-deterministic or not.
291            int: The degree of non-determinism of the soliton automaton.
292            dict: All states of the soliton graph plus all traversals that can be found in each state plus number of traversals found for each burst.
293        """
294        initial_matrix = nx.to_numpy_array(self.soliton_graph.graph)
295        states = [self.soliton_graph] # stores all possible states as soliton graphs, needed to iterate over states 
296        states_plus_traversals = {self.matrix_to_string(initial_matrix): [self.soliton_graph, [], []]} # stores all states plus all traversals that can be found in that state plus number of traversals found for each burst
297        deterministic = True
298        strongly_deterministic = True
299        reachability_deterministic = True
300        degree_of_nondeterminism = 1
301        state_sucstate_burst = dict() # dictionary with number of found traversals for each state + successor state + burst
302        state_sucstate = dict() # dictionary with bool for each state + successor state (first set to False and then only set to True if there is a burst such that there is only one traversal between these states)
303
304        for state in states: # for all states/ soliton graphs of the automaton
305            state_matrix_id = self.matrix_to_string(nx.to_numpy_array(state.graph))
306            all_traversals = []
307            num_traversals_per_burst = []
308            for burst_dict in self.bursts_dicts: # loop over all bursts
309                suc_states_matrix_ids = []
310                traversals = self.call_find_all_travs_given_burst(burst_dict, state) # find traversals with all bursts
311                num_traversals_per_burst.append(len(traversals))
312                first_real_trav = -1
313                maybe_imperf = []
314                real_travs_this_nodepair = []
315                for t, traversal in enumerate(traversals):
316                    if isinstance(traversal, Traversal): # if traversal is a real traversal
317                        all_traversals.append(traversal)
318                        real_travs_this_nodepair.append(traversal)
319                        if first_real_trav == -1:
320                            first_real_trav = t
321                        resulting_matrix = traversal.adjacency_matrices_list[len(traversal.adjacency_matrices_list)-1] # adjacency matrix of the soliton graph the traversal results in
322                        resulting_matrix_id = self.matrix_to_string(resulting_matrix)
323                        if resulting_matrix_id not in states_plus_traversals: # if we found a new state
324                            states_plus_traversals[resulting_matrix_id] = [traversal.resulting_soliton_graph, [], []] # add it to the dictionary
325                            states.append(traversal.resulting_soliton_graph)
326                        if resulting_matrix_id not in suc_states_matrix_ids: # if we found a new state
327                                suc_states_matrix_ids.append(resulting_matrix_id)
328                                if len(suc_states_matrix_ids) > degree_of_nondeterminism:
329                                    degree_of_nondeterminism = len(suc_states_matrix_ids)
330                        if np.array_equal(resulting_matrix, traversals[first_real_trav].adjacency_matrices_list[len(traversals[first_real_trav].adjacency_matrices_list)-1]) == False:
331                                deterministic = False # two or more different soliton graphs have emerged although the same burst was used
332                                reachability_deterministic = False
333                        if (state_matrix_id, resulting_matrix_id, str(list(burst_dict.values()))) not in state_sucstate_burst: # use str(list(...)) since dictionaries are not hashable
334                            state_sucstate_burst[(state_matrix_id, resulting_matrix_id, str(list(burst_dict.values())))] = 1
335                        else: state_sucstate_burst[(state_matrix_id, resulting_matrix_id, str(list(burst_dict.values())))] += 1
336                        state_sucstate[(state_matrix_id, resulting_matrix_id)] = False # put in an entry in the dictionary for these two states
337                    else:
338                        maybe_imperf.append([traversal, state_matrix_id, str(list(burst_dict.values()))]) # this uncompleted traversal between these two states and with this burst may be an imperfect traversal (or no traversal at all)
339                imperf_found, state_sucstate_burst = self.identify_imperf_travs(real_travs_this_nodepair, maybe_imperf, state_sucstate_burst)
340                if (len(traversals) - len(maybe_imperf)) > 1:
341                        strongly_deterministic = False # more than one traversal was found for one burst
342                elif (len(traversals) - len(maybe_imperf)) == 1 and imperf_found:
343                        strongly_deterministic = False # we have only one perfect traversal but at least one imperfect traversal as well
344            states_plus_traversals[state_matrix_id][1] = all_traversals # add all found traversals in this state
345            states_plus_traversals[state_matrix_id][2] = num_traversals_per_burst # add number of traversals found for each burst (in this state)
346
347        for key in state_sucstate_burst:
348            if state_sucstate_burst[key] == 1:
349                state_sucstate[key[0], key[1]] = True # there is a burst such that these is only one traversal between this state and this successor state
350        for key in state_sucstate:
351            if state_sucstate[key] == False: # no such burst exists for these two states
352                reachability_deterministic = False
353                break
354
355        return deterministic, strongly_deterministic, reachability_deterministic, degree_of_nondeterminism, states_plus_traversals

358    def identify_imperf_travs(self, real_travs: list, maybe_imperf: list, state_sucstate_burst: dict):
359        """Uses the list of perfect traversals and a list of possible imperfect traversals for a state and a burst to identify which traversals are actually imperfect traversals. 
360
361        Args:
362            real_travs (list): All real/ perfect traversals that were found.
363            maybe_imperf (list): All traversals that might be imperfect configuration trails.
364            state_sucstate_burst (dict): Dictionary that contains a triple of state, successor state and a burst as key and the number of found traversals between the two states with the burst as value.
365
366        Returns:
367            bool: Whether an imperfect path was found.
368            dict: Modified `state_sucstate_burst`; if imperfect path was found then number of found traversals for state, successor state and burst was raised.
369        """
370        imperf_found = False
371        for candidate in maybe_imperf:
372            for trav in real_travs:
373                if candidate[0][0].pos_and_bindings == trav.pos_and_bindings[0:len(candidate[0][0].pos_and_bindings)]:
374                    imperf_found = True # we found an imperfect path
375                    resulting_matrix = trav.adjacency_matrices_list[len(trav.adjacency_matrices_list)-1] # adjacency matrix of the soliton graph the traversal results in
376                    resulting_matrix_id = self.matrix_to_string(resulting_matrix)
377                    state_sucstate_burst[(candidate[1], resulting_matrix_id, candidate[2])] += 1 # an imperfect traversal between these two states and with this burst was found
378        return imperf_found, state_sucstate_burst

 76            soliton += 1
 77            burst_dict[soliton] = [self.soliton_graph.exterior_nodes_reverse[all_nums[1]], self.soliton_graph.exterior_nodes_reverse[all_nums[2]], entry] # take node ids, not node labels
 78            burst_copy = burst_copy[found.span()[1]:] # remove from burst
 79

123    def find_all_travs_given_burst (self, graph: nx.Graph, burst_dict: dict, t: int, s_pos_all_timesteps: list, bindings_all_timesteps: list, binds_all_timesteps: list, poss_suc_combs_all_timesteps: list, travs: list):
124        """Finds all possible traversals for a given burst by using a recursive backtracking algorithm.
125
126        Args:
127            graph (nx.Graph): Graph the traversals should be found in.
128            burst_dict (dict): The burst that is used. 
129            t (int): Current timestep.
130            s_pos_all_timesteps (list): Soliton positions for all past timesteps.
131            bindings_all_timesteps (list): Binding types of all edges in the graph for all past timesteps.
132            binds_all_timesteps (list): Binding type of the last edge that was traversed by each soliton for all past timesteps.
133            poss_suc_combs_all_timesteps (list): Possible combinations of successor nodes for all past timesteps.
134            travs (list): All currently found traversals.
135
136        Returns:
137            list: All found traversals (is empty if no traversal exists).
138        """
139        t += 1
140        akt_bindings = bindings_all_timesteps[t-1]
141        akt_binds = binds_all_timesteps[t-1]
142        akt_positions = s_pos_all_timesteps[t-1]
143
144        # in this timestep find possible edges for each soliton individually
145        possible_edges = {} # possible edges for each soliton in this timestep
146        possible_nodes = {} # possible nodes for each soliton in this timestep
147        for soliton in burst_dict:
148            end = burst_dict[soliton][1] # end node for this soliton
149            akt_pos = akt_positions[soliton] # current position of this soliton
150            possible_edges[soliton] = []
151            possible_nodes[soliton] = []
152            if akt_pos == -2 or (akt_pos == end and s_pos_all_timesteps[t-2][soliton] != -1 and t > 1): # if soliton already traversed the graph successfully
153                possible_edges[soliton].append(-2)
154                possible_nodes[soliton].append(-2)
155            elif burst_dict[soliton][2] > t: # if soliton's entry time has not come yet
156                possible_edges[soliton].append(-1)
157                possible_nodes[soliton].append(-1)
158            elif burst_dict[soliton][2] == t: # if soliton enters the graph now (place it at its starting node)
159                possible_edges[soliton].append(burst_dict[soliton][0])
160                possible_nodes[soliton].append(burst_dict[soliton][0])
161            else:
162                for node in list(nx.neighbors(graph, akt_pos)): # iterate over all nodes that are adjacent to current position
163                    # soliton can not go to an exterior node that is not the end node, soliton is not allowed to make a direct turnaround and edge to next node has to have the right binding type
164                    if ((node not in self.soliton_graph.exterior_nodes) or (node == end)) and node != s_pos_all_timesteps[t-2][soliton] and akt_bindings[tuple(sorted((akt_pos, node)))] != akt_binds[soliton]:
165                        possible_edges[soliton].append(tuple(sorted((akt_pos, node))))
166                        possible_nodes[soliton].append(node)
167        # find all possible combinations of the found edges 
168        edges_combs = []
169        nodes_combs = []
170        for soliton in possible_edges:
171            if possible_edges[soliton] == []: # if for one soliton there are no possible next edges, no combinations are possible
172                edges_combs.clear()
173                nodes_combs.clear()
174            if soliton != 1 and not any(edges_combs): # if no possible edge combination exists anymore, terminate the loop (not any(edges_combs) makes sure list is empty or only contains empty lists)
175                break
176            for i, edge in enumerate(possible_edges[soliton]): # for all possible edges for this soliton
177                if soliton == 1:
178                    edges_combs.append([edge]) # simply add the edge if it's the first soliton
179                    nodes_combs.append({soliton: possible_nodes[soliton][i]}) # add dictionary with node for this soliton
180                else:
181                    for j, comb in enumerate(edges_combs): # if it's not the first soliton loop over all existing combinations
182                        if edge == -1 or edge == -2 or edge not in comb: # if edge is not already in combination (or soliton is outside the graph)
183                            if len(comb) < soliton:
184                                edges_combs[j].append(edge) # add the edge if this combination doesn't contain an edge for this soliton already
185                                nodes_combs[j][soliton] = possible_nodes[soliton][i]
186                            else: # else copy the combination, remove the last edge (the edge for this soliton), add the edge and add this new combination
187                                comb_copy = copy.deepcopy(comb)
188                                comb_copy.pop()
189                                comb_copy.append(edge)
190                                edges_combs.append(comb_copy)
191                                node_comb_copy = copy.deepcopy(nodes_combs[j])
192                                node_comb_copy[soliton] = possible_nodes[soliton][i] # overwrite
193                                nodes_combs.append(node_comb_copy)
194                        if i == len(possible_edges[soliton])-1: # if we are looking at the last possible edge of this soliton
195                            if len(comb) < soliton: # and the combination does not contain an edge for each soliton
196                                edges_combs[j] = [] # "delete" the combination
197                                nodes_combs[j] = {}
198        edges_combs = [elem for elem in edges_combs if elem != []]
199        nodes_combs = [elem for elem in nodes_combs if elem]
200
201        finished = True
202        for soliton in akt_positions:
203            if (akt_positions[soliton] != burst_dict[soliton][1] and akt_positions[soliton] != -2) or s_pos_all_timesteps[t-2][soliton] == -1 or t < 2: # if not all solitons reached their end node or already left the graph
204                finished = False
205                break
206        # base case 1: if all solitons traversed the graph successfully
207        if finished == True:
208            trav = []
209            for i, pos in enumerate(s_pos_all_timesteps):
210                this_timestep = (pos, bindings_all_timesteps[i]) # for each timestep, add a tuple of the solitons' positions and the bindings
211                trav.append(this_timestep)
212            travs.append(trav) # trav = traversal of all solitons through the graph
213            return travs
214
215        # base case 2: if we have two successor-equivalent configurations in the configuration trail
216        count = 1
217        for k in range(0, len(bindings_all_timesteps)-1):
218            if akt_bindings == bindings_all_timesteps[k] and akt_positions == s_pos_all_timesteps[k] and nodes_combs == poss_suc_combs_all_timesteps[k] and -1 not in nodes_combs[0].values(): # if we already had that exact graph, position map and successor position combinations (and all solitons already entered the graph) in this configuration trail
219                count += 1
220                if count == 2:
221                    trav = []
222                    for i, pos in enumerate(s_pos_all_timesteps):
223                        this_timestep = (pos, bindings_all_timesteps[i])
224                        trav.append(this_timestep)
225                    travs.append([trav[:k+1], k])
226                    return travs
227
228        # iterate over all possible combinations
229        for i, comb in enumerate(nodes_combs):
230            s_pos_all_timesteps.append(comb)
231            bindings = copy.deepcopy(akt_bindings)
232            binds = copy.deepcopy(akt_binds)
233            for j, edge in enumerate(edges_combs[i]): # for all edges in chosen combination
234                if isinstance(edge, tuple): # binds and bindings only change if "edge" is actually an edge (if edge is -1, -2 or the start node, no edge is traversed)
235                    binds[j+1] = bindings[tuple(sorted(edge))] # change bind to binding type of edge that was just traversed
236                    bindings = self.change_bindings(bindings, tuple(sorted(edge))) # change bindings
237            bindings_all_timesteps.append(bindings)
238            binds_all_timesteps.append(binds)
239            poss_suc_combs_all_timesteps.append(nodes_combs)
240            # call function recursively: if eventually all solitons reach their end node if we go further with the decision we just made then travs is changed (new traversal added)
241            travs = self.find_all_travs_given_burst(graph, burst_dict, t, s_pos_all_timesteps, bindings_all_timesteps, binds_all_timesteps, poss_suc_combs_all_timesteps, travs)
242            # try different decisions next, so we need variables in the state before the last decision
243            s_pos_all_timesteps.pop()
244            bindings_all_timesteps.pop()
245            binds_all_timesteps.pop()
246            poss_suc_combs_all_timesteps.pop()
247
248        return travs # if at some point no other traversal could be added, then all possible traversals are found

251    def call_find_all_travs_given_burst (self, burst_dict: dict, soliton_graph: SolitonGraph):
252        """Initializes some parameters and then calls `find_all_travs_given_burst` with them.
253
254        Args:
255            burst_dict (dict): The burst that is used. 
256            soliton_graph (SolitonGraph): Soliton graph the traversals should be found in.
257
258        Returns:
259            list: All found travs as traversals (returns empty list when no traversal is found).
260        """
261        soliton_graph_copy = copy.deepcopy(soliton_graph) # working on copy of graph so no unwanted changes are made
262        graph = soliton_graph_copy.graph
263        t = 0
264        bindings_all_timesteps = [soliton_graph_copy.bindings] # already add bindings for timestep 0
265        soliton_positions = {}
266        binds = {}
267        for soliton in burst_dict:
268            binds[soliton] = 0
269            if burst_dict[soliton][2] == 0: # if soliton enters the graph at timestep 0
270                soliton_positions[soliton] = burst_dict[soliton][0] # starts at its start node
271            else:
272                soliton_positions[soliton] = -1 # soliton is not in graph at timestep 0
273        # already add positions and binds for timestep 0
274        s_pos_all_timesteps = [soliton_positions]
275        binds_all_timesteps = [binds]
276        poss_suc_combs_all_timesteps = []
277        travs = []
278
279        travs = self.find_all_travs_given_burst(graph, burst_dict, t, s_pos_all_timesteps, bindings_all_timesteps, binds_all_timesteps, poss_suc_combs_all_timesteps, travs)
280        traversals = self.build_traversals(travs, soliton_graph)
281
282        return traversals

285    def all_traversals (self):
286        """Calls `call_find_all_travs_given_burst` for all states of the automaton and all bursts in order to get all possible traversals in all states.
287
288        Returns:
289            bool: Whether the soliton automaton is deterministic or not.
290            bool: Whether the soliton automaton is strongly deterministic or not.
291            bool: Whether the soliton automaton is reachability-deterministic or not.
292            int: The degree of non-determinism of the soliton automaton.
293            dict: All states of the soliton graph plus all traversals that can be found in each state plus number of traversals found for each burst.
294        """
295        initial_matrix = nx.to_numpy_array(self.soliton_graph.graph)
296        states = [self.soliton_graph] # stores all possible states as soliton graphs, needed to iterate over states 
297        states_plus_traversals = {self.matrix_to_string(initial_matrix): [self.soliton_graph, [], []]} # stores all states plus all traversals that can be found in that state plus number of traversals found for each burst
298        deterministic = True
299        strongly_deterministic = True
300        reachability_deterministic = True
301        degree_of_nondeterminism = 1
302        state_sucstate_burst = dict() # dictionary with number of found traversals for each state + successor state + burst
303        state_sucstate = dict() # dictionary with bool for each state + successor state (first set to False and then only set to True if there is a burst such that there is only one traversal between these states)
304
305        for state in states: # for all states/ soliton graphs of the automaton
306            state_matrix_id = self.matrix_to_string(nx.to_numpy_array(state.graph))
307            all_traversals = []
308            num_traversals_per_burst = []
309            for burst_dict in self.bursts_dicts: # loop over all bursts
310                suc_states_matrix_ids = []
311                traversals = self.call_find_all_travs_given_burst(burst_dict, state) # find traversals with all bursts
312                num_traversals_per_burst.append(len(traversals))
313                first_real_trav = -1
314                maybe_imperf = []
315                real_travs_this_nodepair = []
316                for t, traversal in enumerate(traversals):
317                    if isinstance(traversal, Traversal): # if traversal is a real traversal
318                        all_traversals.append(traversal)
319                        real_travs_this_nodepair.append(traversal)
320                        if first_real_trav == -1:
321                            first_real_trav = t
322                        resulting_matrix = traversal.adjacency_matrices_list[len(traversal.adjacency_matrices_list)-1] # adjacency matrix of the soliton graph the traversal results in
323                        resulting_matrix_id = self.matrix_to_string(resulting_matrix)
324                        if resulting_matrix_id not in states_plus_traversals: # if we found a new state
325                            states_plus_traversals[resulting_matrix_id] = [traversal.resulting_soliton_graph, [], []] # add it to the dictionary
326                            states.append(traversal.resulting_soliton_graph)
327                        if resulting_matrix_id not in suc_states_matrix_ids: # if we found a new state
328                                suc_states_matrix_ids.append(resulting_matrix_id)
329                                if len(suc_states_matrix_ids) > degree_of_nondeterminism:
330                                    degree_of_nondeterminism = len(suc_states_matrix_ids)
331                        if np.array_equal(resulting_matrix, traversals[first_real_trav].adjacency_matrices_list[len(traversals[first_real_trav].adjacency_matrices_list)-1]) == False:
332                                deterministic = False # two or more different soliton graphs have emerged although the same burst was used
333                                reachability_deterministic = False
334                        if (state_matrix_id, resulting_matrix_id, str(list(burst_dict.values()))) not in state_sucstate_burst: # use str(list(...)) since dictionaries are not hashable
335                            state_sucstate_burst[(state_matrix_id, resulting_matrix_id, str(list(burst_dict.values())))] = 1
336                        else: state_sucstate_burst[(state_matrix_id, resulting_matrix_id, str(list(burst_dict.values())))] += 1
337                        state_sucstate[(state_matrix_id, resulting_matrix_id)] = False # put in an entry in the dictionary for these two states
338                    else:
339                        maybe_imperf.append([traversal, state_matrix_id, str(list(burst_dict.values()))]) # this uncompleted traversal between these two states and with this burst may be an imperfect traversal (or no traversal at all)
340                imperf_found, state_sucstate_burst = self.identify_imperf_travs(real_travs_this_nodepair, maybe_imperf, state_sucstate_burst)
341                if (len(traversals) - len(maybe_imperf)) > 1:
342                        strongly_deterministic = False # more than one traversal was found for one burst
343                elif (len(traversals) - len(maybe_imperf)) == 1 and imperf_found:
344                        strongly_deterministic = False # we have only one perfect traversal but at least one imperfect traversal as well
345            states_plus_traversals[state_matrix_id][1] = all_traversals # add all found traversals in this state
346            states_plus_traversals[state_matrix_id][2] = num_traversals_per_burst # add number of traversals found for each burst (in this state)
347
348        for key in state_sucstate_burst:
349            if state_sucstate_burst[key] == 1:
350                state_sucstate[key[0], key[1]] = True # there is a burst such that these is only one traversal between this state and this successor state
351        for key in state_sucstate:
352            if state_sucstate[key] == False: # no such burst exists for these two states
353                reachability_deterministic = False
354                break
355
356        return deterministic, strongly_deterministic, reachability_deterministic, degree_of_nondeterminism, states_plus_traversals

125

128            burst_dict (dict): The burst that is used. 
129            t (int): Current timestep.
130            s_pos_all_timesteps (list): Soliton positions for all past timesteps.
131            bindings_all_timesteps (list): Binding types of all edges in the graph for all past timesteps.
132            binds_all_timesteps (list): Binding type of the last edge that was traversed by each soliton for all past timesteps.
133            poss_suc_combs_all_timesteps (list): Possible combinations of successor nodes for all past timesteps.
134            travs (list): All currently found traversals.
135
136        Returns:
137            list: All found traversals (is empty if no traversal exists).
138        """
139        t += 1
140        akt_bindings = bindings_all_timesteps[t-1]
141        akt_binds = binds_all_timesteps[t-1]
142        akt_positions = s_pos_all_timesteps[t-1]
143
144        # in this timestep find possible edges for each soliton individually
145        possible_edges = {} # possible edges for each soliton in this timestep
146        possible_nodes = {} # possible nodes for each soliton in this timestep
147        for soliton in burst_dict:
148            end = burst_dict[soliton][1] # end node for this soliton
149            akt_pos = akt_positions[soliton] # current position of this soliton
150            possible_edges[soliton] = []
151            possible_nodes[soliton] = []
152            if akt_pos == -2 or (akt_pos == end and s_pos_all_timesteps[t-2][soliton] != -1 and t > 1): # if soliton already traversed the graph successfully
153                possible_edges[soliton].append(-2)
154                possible_nodes[soliton].append(-2)
155            elif burst_dict[soliton][2] > t: # if soliton's entry time has not come yet
156                possible_edges[soliton].append(-1)
157                possible_nodes[soliton].append(-1)
158            elif burst_dict[soliton][2] == t: # if soliton enters the graph now (place it at its starting node)
159                possible_edges[soliton].append(burst_dict[soliton][0])
160                possible_nodes[soliton].append(burst_dict[soliton][0])
161            else:
162                for node in list(nx.neighbors(graph, akt_pos)): # iterate over all nodes that are adjacent to current position
163                    # soliton can not go to an exterior node that is not the end node, soliton is not allowed to make a direct turnaround and edge to next node has to have the right binding type
164                    if ((node not in self.soliton_graph.exterior_nodes) or (node == end)) and node != s_pos_all_timesteps[t-2][soliton] and akt_bindings[tuple(sorted((akt_pos, node)))] != akt_binds[soliton]:
165                        possible_edges[soliton].append(tuple(sorted((akt_pos, node))))
166                        possible_nodes[soliton].append(node)
167        # find all possible combinations of the found edges 
168        edges_combs = []
169        nodes_combs = []
170        for soliton in possible_edges:
171            if possible_edges[soliton] == []: # if for one soliton there are no possible next edges, no combinations are possible
172                edges_combs.clear()
173                nodes_combs.clear()

175                break

177                if soliton == 1:
178                    edges_combs.append([edge]) # simply add the edge if it's the first soliton
179                    nodes_combs.append({soliton: possible_nodes[soliton][i]}) # add dictionary with node for this soliton
180                else:
181                    for j, comb in enumerate(edges_combs): # if it's not the first soliton loop over all existing combinations
182                        if edge == -1 or edge == -2 or edge not in comb: # if edge is not already in combination (or soliton is outside the graph)
183                            if len(comb) < soliton:
184                                edges_combs[j].append(edge) # add the edge if this combination doesn't contain an edge for this soliton already
185                                nodes_combs[j][soliton] = possible_nodes[soliton][i]
186                            else: # else copy the combination, remove the last edge (the edge for this soliton), add the edge and add this new combination
187                                comb_copy = copy.deepcopy(comb)
188                                comb_copy.pop()
189                                comb_copy.append(edge)
190                                edges_combs.append(comb_copy)
191                                node_comb_copy = copy.deepcopy(nodes_combs[j])
192                                node_comb_copy[soliton] = possible_nodes[soliton][i] # overwrite
193                                nodes_combs.append(node_comb_copy)
194                        if i == len(possible_edges[soliton])-1: # if we are looking at the last possible edge of this soliton
195                            if len(comb) < soliton: # and the combination does not contain an edge for each soliton
196                                edges_combs[j] = [] # "delete" the combination
197                                nodes_combs[j] = {}
198        edges_combs = [elem for elem in edges_combs if elem != []]
199        nodes_combs = [elem for elem in nodes_combs if elem]
200
201        finished = True
202        for soliton in akt_positions:
203            if (akt_positions[soliton] != burst_dict[soliton][1] and akt_positions[soliton] != -2) or s_pos_all_timesteps[t-2][soliton] == -1 or t < 2: # if not all solitons reached their end node or already left the graph
204                finished = False
205                break
206        # base case 1: if all solitons traversed the graph successfully
207        if finished == True:

210                this_timestep = (pos, bindings_all_timesteps[i]) # for each timestep, add a tuple of the solitons' positions and the bindings
211                trav.append(this_timestep)
212            travs.append(trav) # trav = traversal of all solitons through the graph
213            return travs
214
215        # base case 2: if we have two successor-equivalent configurations in the configuration trail
216        count = 1
217        for k in range(0, len(bindings_all_timesteps)-1):
218            if akt_bindings == bindings_all_timesteps[k] and akt_positions == s_pos_all_timesteps[k] and nodes_combs == poss_suc_combs_all_timesteps[k] and -1 not in nodes_combs[0].values(): # if we already had that exact graph, position map and successor position combinations (and all solitons already entered the graph) in this configuration trail
219                count += 1
220                if count == 2:
221                    trav = []
222                    for i, pos in enumerate(s_pos_all_timesteps):
223                        this_timestep = (pos, bindings_all_timesteps[i])
224                        trav.append(this_timestep)
225                    travs.append([trav[:k+1], k])
226                    return travs
227
228        # iterate over all possible combinations
229        for i, comb in enumerate(nodes_combs):
230            s_pos_all_timesteps.append(comb)
231            bindings = copy.deepcopy(akt_bindings)
232            binds = copy.deepcopy(akt_binds)
233            for j, edge in enumerate(edges_combs[i]): # for all edges in chosen combination
234                if isinstance(edge, tuple): # binds and bindings only change if "edge" is actually an edge (if edge is -1, -2 or the start node, no edge is traversed)
235                    binds[j+1] = bindings[tuple(sorted(edge))] # change bind to binding type of edge that was just traversed
236                    bindings = self.change_bindings(bindings, tuple(sorted(edge))) # change bindings
237            bindings_all_timesteps.append(bindings)
238            binds_all_timesteps.append(binds)
239            poss_suc_combs_all_timesteps.append(nodes_combs)
240            # call function recursively: if eventually all solitons reach their end node if we go further with the decision we just made then travs is changed (new traversal added)
241            travs = self.find_all_travs_given_burst(graph, burst_dict, t, s_pos_all_timesteps, bindings_all_timesteps, binds_all_timesteps, poss_suc_combs_all_timesteps, travs)
242            # try different decisions next, so we need variables in the state before the last decision
243            s_pos_all_timesteps.pop()
244            bindings_all_timesteps.pop()
245            binds_all_timesteps.pop()
246            poss_suc_combs_all_timesteps.pop()
247
248        return travs # if at some point no other traversal could be added, then all possible traversals are found
